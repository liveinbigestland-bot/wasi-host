```zig
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = .ReleaseSmall;

    const exe = b.addExecutable(.{
        .name = "wasi-host",
        .root_source_file = .{ .path = "main.zig" },
        .target = target,
        .optimize = optimize,
        .strip = true,
    });

    exe.linkLibC();
    exe.addCSourceFiles(.{
        .files = &.{
            "wasm3/src/m3_core.c",
            "wasm3/src/m3_env.c",
            "wasm3/src/m3_exec.c",
            "wasm3/src/m3_load.c",
            "wasm3/src/m3_stack.c",
            "wasm3/src/m3_wasi.c",
        },
        .flags = &.{"-DM3_ENABLE_WASI=1"},
    });
    exe.addIncludePath(.{ .path = "wasm3/include" });

    b.installArtifact(exe);
}
```

### 3. `main.zig`（最终完整版 + CPU使用率 + 宿主信息 + 多插件并发）

zig

```zig
const std = @import("std");
const json = std.json;
const wasm3 = @cImport({
    @cInclude("wasm3.h");
    @cInclude("m3_env.h");
});

// ========== 批量内嵌插件(自行增删即可) ==========
const plug_ai  = @embedFile("plug/ai.wasm");
const plug_api = @embedFile("plug/api.wasm");

const EmbeddedPlug = struct{
    path: []const u8,
    data: []const u8,
};
const embedded_list = [_]EmbeddedPlug{
    .{.path = "plug/ai.wasm", .data = plug_ai},
    .{.path = "plug/api.wasm", .data = plug_api},
};

// ========== 配置结构体 ==========
const PlugConfig = struct{
    name: []const u8,
    embed_path: []const u8,
    mem_kb: u32,
    timeout_ms: u32,
    network: bool,
    write: bool,
    allow_host_info: bool,
};
const AppConfig = struct{
    plugins: []PlugConfig,
};

// ========== 宿主机硬件信息采集 ==========
fn readCpuUsage() f32 {
    if (@import("builtin").os.tag != .linux) return 0.0;

    var f = std.fs.openFileAbsolute("/proc/stat", .{}) catch return 0.0;
    defer f.close();

    var buf: [256]u8 = undefined;
    const n = f.read(&buf) catch return 0.0;
    const line = buf[0..n];

    var iter = std.mem.tokenizeAny(u8, line, " \n");
    var sum: u64 = 0;
    var idle: u64 = 0;
    var idx: u8 = 0;

    while (iter.next()) |tok| : (idx += 1) {
        const val = std.fmt.parseUnsigned(u64, tok, 10) catch 0;
        sum += val;
        if (idx == 4) idle = val;
        if (idx > 7) break;
    }
    if (sum == 0) return 0.0;
    return 100.0 - (@as(f32, idle) / @as(f32, sum) * 100.0);
}

// 对外导出给WASM插件调用
export fn host_cpu_cores() u32 {
    return std.Thread.cpuCount() orelse 1;
}
export fn host_cpu_usage() f32 {
    return readCpuUsage();
}
export fn host_mem_total() u64 {
    return std.os.memTotal() catch 0;
}
export fn host_mem_free() u64 {
    return std.os.memFree() catch 0;
}
export fn host_os_tag() [*:0]const u8 {
    comptime var os_name: []const u8 = @tagName(@import("builtin").os.tag);
    return os_name.ptr;
}

// ========== 单插件独立运行逻辑 ==========
fn runPlugin(alloc: std.mem.Allocator, cfg: PlugConfig) void {
    std.debug.print("\n[启动插件] {s}\n", .{cfg.name});

    const env = wasm3.m3_NewEnvironment() orelse return;
    defer wasm3.m3_FreeEnvironment(env);

    const rt = wasm3.m3_NewRuntime(env, cfg.mem_kb * 1024, null) orelse return;
    defer wasm3.m3_FreeRuntime(rt);

    // 匹配内嵌插件
    const wasm_bin = blk: {
        for (embedded_list) |e| {
            if (std.mem.eql(u8, e.path, cfg.embed_path)) {
                break :blk e.data;
            }
        }
        std.debug.print("[错误] 未找到插件 {s}\n", .{cfg.embed_path});
        return;
    };

    // 加载模块
    var mod: ?*wasm3.M3Module = null;
    if (wasm3.m3_ParseModule(env, &mod, wasm_bin.ptr, wasm_bin.len) != 0) return;
    if (wasm3.m3_LoadModule(rt, mod) != 0) return;

    // 绑定宿主硬件API（按需开关）
    if (cfg.allow_host_info) {
        _ = wasm3.m3_LinkRawFunction(rt, "host", "cpu_cores",  "i()", host_cpu_cores);
        _ = wasm3.m3_LinkRawFunction(rt, "host", "cpu_usage",  "f()", host_cpu_usage);
        _ = wasm3.m3_LinkRawFunction(rt, "host", "mem_total",  "I()", host_mem_total);
        _ = wasm3.m3_LinkRawFunction(rt, "host", "mem_free",  "I()", host_mem_free);
        _ = wasm3.m3_LinkRawFunction(rt, "host", "os_tag",  "*()", host_os_tag);
    }

    // WASI 权限隔离
    var wasi_cfg = wasm3.M3WasiConfig{};
    wasi_cfg.allow_network = if(cfg.network) 1 else 0;
    wasi_cfg.allow_filesystem_write = if(cfg.write) 1 else 0;
    wasi_cfg.allow_env_vars = 0;
    _ = wasm3.m3_LinkWasiCore(rt, wasi_cfg);

    // 执行入口
    const start_fn = wasm3.m3_FindFunction(rt, "_start") orelse return;
    _ = wasm3.m3_Call(start_fn, 0, null, null);

    std.debug.print("[完成插件] {s}\n", .{cfg.name});
}

// ========== 入口 ==========
pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    const alloc = gpa.allocator();

    // 加载JSON配置
    const cfg_text = try std.fs.cwd().readFileAlloc(alloc, "config.json", 1024 * 64);
    defer alloc.free(cfg_text);
    const parsed = try json.parseFromSlice(AppConfig, alloc, cfg_text, .{});
    defer parsed.deinit();
    const app_cfg = parsed.value;

    // 多线程并行所有插件
    for (app_cfg.plugins) |plug_cfg| {
        _ = try std.Thread.spawn(.{}, runPlugin, .{alloc, plug_cfg});
    }

    // 常驻
    while(true) std.time.sleep(500 * std.time.ns_per_ms);
}
```

## 三、一键打包脚本（3平台）

新建 `build-all.sh`

bash

运行

```bash
#!/bin/bash
mkdir -p output

# Linux x86_64
zig build -Doptimize=ReleaseSmall
cp zig-out/bin/wasi-host output/wasi-host-linux-x64

# Windows x64
zig build -Dtarget=x86_64-windows-gnu -Doptimize=ReleaseSmall
cp zig-out/bin/wasi-host.exe output/wasi-host-win-x64.exe

# ARM64 服务器
zig build -Dtarget=aarch64-linux-gnu -Doptimize=ReleaseSmall
cp zig-out/bin/wasi-host output/wasi-host-linux-arm64

echo "✅ 全平台打包完成，产物目录：output/"
```

授权运行：

bash

运行

```bash
chmod +x build-all.sh
./build-all.sh
```
